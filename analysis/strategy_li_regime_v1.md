### 卖点逻辑

```python
def sell_li_regime_v1(buy_idx, td_seg, max_end):
    """离 regime v1 卖点: bull 单独"""
    cross_count = 0
    running_max = td_seg[buy_idx]

    for k in range(buy_idx + 1, max_end + 1):
        if not isnan(td_seg[k]):
            running_max = max(running_max, td_seg[k])

        if k > 0 and running_max >= 89 \
           and td_seg[k] < 89 and td_seg[k-1] >= 89:
            cross_count += 1
            if cross_count == 2:
                return k, 'bull_2nd'

    return max_end, 'timeout'
```

## 11. 实施伪代码 (买卖完整版)

```python
def li_regime_buy_signal(stock, day, market):
    """v1 买点判断 (3 道关卡)"""
    # Gate 1: regime
    if market.y_gua(day) != '101': return None  # 离

    # Gate 2: 时机 (用坤, 跟兑 regime 一致)
    if stock.d_gua(day) != '000': return None  # 坤

    # Gate 3: 卦象避雷
    if market.d_gua(day) == '101': return None  # 大盘日离 (强)
    if stock.m_gua(day) in {'011', '001', '101'}: return None  # 个股月: 巽/艮/离
    if stock.y_gua(day) == '011': return None  # 个股年巽

    return {
        'code': stock.code,
        'date': day,
        'price': stock.close(day),
        # 备选加分: 命中股m=110 兑 时仓位 1.5×
        'bonus': 0.5 if stock.m_gua(day) == '110' else 0.0,
    }


def li_regime_sell_signal(stock, buy_idx, hold_max=60):
    """v1 卖点判断 (bull 单独)"""
    cross_count = 0
    running_max = stock.d_trend(buy_idx)
    end_idx = buy_idx + hold_max

    for k in range(buy_idx + 1, end_idx + 1):
        td_k = stock.d_trend(k)
        if not isnan(td_k):
            running_max = max(running_max, td_k)

        td_prev = stock.d_trend(k - 1)
        if running_max >= 89 and td_k < 89 and td_prev >= 89:
            cross_count += 1
            if cross_count == 2:
                return k, 'bull_2nd', stock.close(k)

    return end_idx, 'timeout', stock.close(end_idx)
```
