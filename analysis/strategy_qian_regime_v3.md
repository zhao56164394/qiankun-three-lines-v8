def sell_qian_regime_v3(buy_idx, td_seg, max_end):
    """乾 regime v3 卖点: bull 单独"""
    cross_count = 0
    running_max = td_seg[buy_idx]

    for k in range(buy_idx + 1, max_end + 1):
        if not isnan(td_seg[k]):
            running_max = max(running_max, td_seg[k])

        if k > 0 and running_max >= 89 \
           and td_seg[k] < 89 and td_seg[k-1] >= 89:

            if cross_count == 2:
                return k, 'bull_2nd'

### w6 的特征 — 抱团破灭后震荡

w6 大盘月卦分布:
```
mkt_m=111 乾: 41.3%
mkt_m=101 离: 43.3%  ← 离比乾还多!
```

对比强 alpha 段:
```
w3_2020: mkt_m=乾 65.4%, 离 24.0%  (典型抱团)
w7_2025_26: mkt_m=乾 47.8%, 离 15.3%  (抱团延续)
```

**业务理解**: 大盘 y_gua=乾 是慢指标 (年线), 滞后于大盘 m_gua (月线)。w6 段的大盘月卦已经从抱团转向"冲高末段", 但年线还显示乾 — 这是**年线滞后真相**导致的 regime 误判。

### 投产规则 (添加段层面环境检查)

**推荐**: 投产时**不仅看大盘 y_gua=乾**, 还要看**最近 20 个交易日大盘 m_gua 中"乾占比 / (乾+离)"**:

```python
def is_qian_regime_alive(today, last_20_days):
    """判断当前乾 regime 是否真活, 还是 w6 类型死局"""
    if today.market_y_gua != '111': return False
    
    last_20 = last_20_days.market_m_gua
    qian_ratio = (last_20 == '111').sum()
    li_ratio = (last_20 == '101').sum()
    
    # w6 特征: 离 ≥ 乾
    if li_ratio >= qian_ratio:
        return False  # 抱团破灭
    
    return True
```

## 12. 实施伪代码 (买卖完整版)

```python
def qian_regime_buy_signal(stock, day, market):
    """v3 买点判断 (5 道关卡)"""
    # Gate 1: regime
    if market.y_gua(day) != '111': return None  # 乾

    # Gate 0 推荐: 段层面活性检查 (避 w6 死局)
    last_20 = market.m_gua_last_n(day, 20)
    qian_cnt = (last_20 == '111').sum()
    li_cnt = (last_20 == '101').sum()
    if li_cnt >= qian_cnt: return None  # 抱团破灭信号

    # Gate 2: 时机
    if stock.d_gua(day) != '011': return None  # 巽

    # Gate 3: 卦象避雷
    if market.d_gua(day) in ['100', '101', '110']: return None
    if market.m_gua(day) == '101': return None
    if stock.m_gua(day) in ['100', '101']: return None

    # Gate 4: 涨幅避雷
    ret_10d = (stock.close(day) / stock.close(day - 10) - 1) * 100
    if ret_10d > 15: return None

    # Gate 5: 软排名 score
    score = 0
    if stock.m_gua(day) == '010': score += 1   # 坎
    if stock.y_gua(day) == '010': score += 1   # 坎

    if score < 1: return None

    return {
        'code': stock.code,
        'date': day,
        'price': stock.close(day),
        'score': score,
    }


def qian_regime_sell_signal(stock, buy_idx, hold_max=60):
    """v3 卖点判断 (bull 单独)"""
    cross_count = 0
    running_max = stock.d_trend(buy_idx)
    end_idx = buy_idx + hold_max

    for k in range(buy_idx + 1, end_idx + 1):
        td_k = stock.d_trend(k)
        if not isnan(td_k):
            running_max = max(running_max, td_k)

        td_prev = stock.d_trend(k - 1)
        if running_max >= 89 and td_k < 89 and td_prev >= 89:
            cross_count += 1
            if cross_count == 2:
                return k, 'bull_2nd', stock.close(k)

    return end_idx, 'timeout', stock.close(end_idx)
```
